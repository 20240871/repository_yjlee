#include <stdio.h>

int main(void)
{
    int ch; // 입력 문자를 저장할 변수

    ch = getchar(); // 함수가 반환하는 문자를 바로 저장
    printf("입력한 문자: ");
    putchar(ch); // 입력한 문자 출력
    putchar('\n'); // 개행 문자 출력

    char small, cap = 'G'; // char형 변수 선언과 초기화

    if ((cap >= 'A') && (cap <= 'Z')) // 대문자 범위라면
    {
        small = cap + ('a' - 'A'); // 대/소문자의 차이를 더해 소문자로 변환
    }
    printf("대문자: %c %c", cap, '\n'); // '\n'를 %c로 출력하면 줄이 바뀜
    printf("소문자: %c", small);

    return 0;
}
